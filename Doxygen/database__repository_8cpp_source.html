125int DatabaseRepository::registerUser(const std::string& username, const std::string& password) const {
126 // Проверка на пустые имя пользователя и пароль.
127 if (username.empty() || password.empty())
128 return -1;
129
130 // Хеширование пароля.
131 QByteArray passwordData(password.c_str(), password.size());
132 QByteArray passwordHash = QCryptographicHash::hash(passwordData, QCryptographicHash::Sha256).toHex();
133
134 // Проверка на существование пользователя с таким же именем.
135 QSqlQuery query_check;
136 query_check.prepare("SELECT * FROM Users WHERE username = :username");
137 query_check.bindValue(":username", QString::fromStdString(username));
138 if (query_check.exec() && query_check.next())
139 return -1;
140
141 // Вставка нового пользователя в таблицу Users.
142 QSqlQuery query;
143 query.prepare("INSERT INTO Users(username, password_hash) VALUES (:username, :password_hash)");
144 query.bindValue(":username", QString::fromStdString(username));
145 query.bindValue(":password_hash", QString::fromStdString(std::string(passwordHash.constData(), passwordHash.size())));
146
147 // Если запрос выполнен успешно, возвращаем идентификатор пользователя.
148 if (query.exec())
149 return query.lastInsertId().toInt();
150
151 return -1;
152}
